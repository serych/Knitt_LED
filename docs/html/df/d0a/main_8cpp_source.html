124static bool ensureNvs() {
125 esp_err_t err = nvs_flash_init();
126 if (err == ESP_ERR_NVS_NO_FREE_PAGES || err == ESP_ERR_NVS_NEW_VERSION_FOUND) {
127 Serial.printf("NVS init issue: %s, erasing...\n", esp_err_to_name(err));
128 nvs_flash_erase();
129 err = nvs_flash_init();
130 }
131 if (err != ESP_OK) {
132 Serial.printf("NVS init failed: %s\n", esp_err_to_name(err));
133 return false;
134 }
135 Serial.println("NVS init ok");
136 return true;
137}

148// Wrap row index 0..h-1
149static int wrapRow(int r) {
150 if (pattern.h <= 0) return 0;
151 while (r < 0) r += pattern.h;
152 while (r >= pattern.h) r -= pattern.h;
153 return r;
154}
155
156// Map "step +1/-1" to internal row index change based on direction
157// rowFromBottom=false: +1 means go down (row index +1)
158// rowFromBottom=true: +1 means go up in index (row index -1) because counting starts from bottom
159static void stepRow(int step) {
160 if (step > 0 && cfg.blinkWarning && !rowConfirmed[cfg.activeRow]) {
161 cfg.warnBlinkActive = true;
162 } else if (step < 0) {
163 cfg.warnBlinkActive = false;
164 }
165
166 int dir = cfg.rowFromBottom ? -1 : +1;
167 cfg.activeRow = wrapRow(cfg.activeRow + step * dir);
168
169 saveConfig(cfg);
170}

172static int shownRowNumber1based() {
173 // OLED/web should show "Row 1" at the "start" according to direction
174 // activeRow is internal index where 0 is top.
175 return cfg.rowFromBottom ? (pattern.h - cfg.activeRow) : (cfg.activeRow + 1);
176}

286 // ---- Try STA WiFi first ----

289 Serial.printf("WiFi connected, IP: %s\n", ip.c_str());
290 oled.showIp(ip);

304 wifiStartPortal(

313 portalActive = false;
314

317 oled.showIp(ipStr);

319
320 wifiStopPortal(dns);
321 server.stop();

327 }
328 );

364 // ---- Detect changes made by Web UI and refresh outputs ----
365 // Web UI updates cfg.activeRow via /api/row and changes cfg via /api/config.
366 // This observer makes OLED + LEDs follow those changes.
367 static int lastRow = -999;
368 static uint32_t lastTot = 0;
369 static bool lastWarn = false;
370 static uint8_t lastBright = 255;
371 static uint32_t lastCA = 0;
372 static uint32_t lastCC = 0;
373 static uint32_t lastCI = 0;
374 static bool lastRB = false;
375 static int lastH = -1;
376 static int lastW = -1;
377
378 bool changed =
379 (cfg.activeRow != lastRow) ||
380 (cfg.totalPulses != lastTot) ||
381 (cfg.warnBlinkActive != lastWarn) ||
382 (cfg.brightness != lastBright) ||
383 (cfg.colorActive != lastCA) ||
384 (cfg.colorConfirmed != lastCC) ||
385 (cfg.colorInactive != lastCI) ||
386 (cfg.rowFromBottom != lastRB) ||
387 (pattern.h != lastH) ||
388 (pattern.w != lastW);
389
390 if (changed) {
391 // Apply brightness change immediately
392 leds.setBrightness(cfg.brightness);
393
394 refreshOutputs();
395
396 lastRow = cfg.activeRow;
397 lastTot = cfg.totalPulses;
398 lastWarn = cfg.warnBlinkActive;
399 lastBright = cfg.brightness;
400 lastCA = cfg.colorActive;
401 lastCC = cfg.colorConfirmed;
402 lastCI = cfg.colorInactive;
403 lastRB = cfg.rowFromBottom;
404 lastH = pattern.h;
405 lastW = pattern.w;
406 }
407
408 // ---- Blink warning handling ----
409 static uint32_t lastBlinkMs = 0;
410 static bool blinkOn = true;
411
412 if (cfg.warnBlinkActive && cfg.blinkWarning) {
413 uint32_t now = millis();
414 if (now - lastBlinkMs > 300) {
415 lastBlinkMs = now;
416 blinkOn = !blinkOn;
417 leds.blinkRow(pattern, cfg.activeRow, rowConfirmed[cfg.activeRow], cfg, blinkOn);
418 }
419 }
420
421 delay(5);
422}

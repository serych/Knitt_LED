39 uint32_t dimColor(uint32_t packed, uint8_t brightness) {
40 uint8_t a = (packed >> 16) & 0xFF;
41 uint8_t b = (packed >> 8) & 0xFF;
42 uint8_t c = packed & 0xFF;
43 a = (uint8_t)((uint16_t)a * brightness / 255);
44 b = (uint8_t)((uint16_t)b * brightness / 255);
45 c = (uint8_t)((uint16_t)c * brightness / 255);
46 return ((uint32_t)a << 16) | ((uint32_t)b << 8) | c;
47 }

57 void showRow(const Pattern& p, int row, bool confirmed, const AppConfig& cfg) {
58 _strip.clear();
59
60 int w = p.w;
61 int leds = _strip.numPixels();
62 int rowLeds = max(0, leds - 1);
63 int use = min(w, rowLeds);
64 uint32_t baseOn = confirmed ? cfg.colorConfirmed : cfg.colorActive;
65 uint32_t baseOff = cfg.colorInactive;
66 uint32_t colOn = dimColor(baseOn, _rowBrightness);
67 uint32_t colOff = dimColor(baseOff, _rowBrightness);
68
69 for (int c = 0; c < use; c++) {
70 int li = 1 + (w - 1) - c;
71 if (li >= 1 && li < leds) {
72 _strip.setPixelColor(li, p.px[row][c] ? colOn : colOff);
73 }
74 }
75 _strip.setPixelColor(0, _statusColor);
76 _strip.show();
77 }

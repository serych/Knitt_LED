35 // LED0 is RIGHTMOST = needle #1.
36 // Internal col 0 is LEFT, so mapping: col->led = (w-1-col)
37 void showRow(const Pattern& p, int row, bool confirmed, const AppConfig& cfg) {
38 _strip.clear();
39
40 int w = p.w;
41 int leds = _strip.numPixels();
42 int use = min(w, leds);
43 uint32_t col = confirmed ? cfg.colorConfirmed : cfg.colorActive;
44
45 for (int c = 0; c < use; c++) {
46 if (!p.px[row][c]) continue;
47 int li = (w - 1) - c;
48 if (li >= 0 && li < leds) _strip.setPixelColor(li, col);
49 }
50 _strip.show();
51 }

53 void blinkRow(const Pattern& p, int row, bool confirmed, const AppConfig& cfg, bool on) {
54 if (on) showRow(p, row, confirmed, cfg);

58 }

36static String htmlEscape(const String& s) {
37 String out;
38 out.reserve(s.length());
39 for (char c : s) {
40 switch (c) {
41 case '&': out += F("&amp;"); break;
42 case '<': out += F("&lt;"); break;
43 case '>': out += F("&gt;"); break;
44 case '"': out += F("&quot;"); break;
45 case '\'': out += F("&#39;"); break;
46 default: out += c; break;
47 }
48 }
49 return out;
50}

52// Normalize any incoming "file" into an absolute path under /patterns.
53// Examples:
54// - "diamond.json" -> "/patterns/diamond.json"
55// - "/diamond.json" -> "/patterns/diamond.json"
56// - "/patterns/diamond.json" -> "/patterns/diamond.json"
57static String normalizePatternPath(String file) {
58 file.trim();
59 if (file.isEmpty()) return "/patterns/default.json";
60
61 // strip query if accidentally included
62 int q = file.indexOf('?');
63 if (q >= 0) file = file.substring(0, q);
64
65 if (!file.startsWith("/")) {
66 return "/patterns/" + file;
67 }
68
69 if (file.startsWith("/") && !file.startsWith("/patterns/")) {
70 // if it is "/name.json" (single segment), move into /patterns
71 if (file.indexOf('/', 1) < 0) {
72 return "/patterns" + file;
73 }
74 }
75
76 return file;
77}

79static int wrapRowIndex(int r, int h) {
80 if (h <= 0) return 0;
81 while (r < 0) r += h;
82 while (r >= h) r -= h;
83 return r;
84}

86// Step semantics:
87// delta is "next row" (+1) or "previous row" (-1) in the user-selected direction.
88static void stepRowFromWeb(int deltaStep) {
89 int h = D.pattern->h;
90 if (h <= 0) return;
91
92 int dir = D.cfg->rowFromBottom ? -1 : +1;
93 if (deltaStep > 0 && D.cfg->blinkWarning && !D.rowConfirmed[D.cfg->activeRow]) {
94 D.cfg->warnBlinkActive = true;
95 } else if (deltaStep < 0) {
96 D.cfg->warnBlinkActive = false;
97 }
98 D.cfg->activeRow = wrapRowIndex(D.cfg->activeRow + deltaStep * dir, h);
99
100 saveConfig(*D.cfg);
101}

107bool loadPatternFile(const String& pathIn, Pattern& p) {
108 String path = normalizePatternPath(pathIn);
109 if (!LittleFS.exists(path)) return false;
110
111 File f = LittleFS.open(path, "r");
112 if (!f) return false;
113 String json = f.readString();
114 f.close();
115
116 return jsonToPattern(json, p);
117}

119bool savePatternFile(const String& pathIn, const Pattern& p) {
120 String path = normalizePatternPath(pathIn);
121 File f = LittleFS.open(path, "w");
122 if (!f) return false;
123 f.print(patternToJson(p));
124 f.close();
125 return true;
126}

163

167
168 if (!fname.endsWith(".json")) fname += ".json";
169 String path = "/patterns/" + fname;
170
171 uploadFile = LittleFS.open(path, "w");
172 }
173 else if (up.status == UPLOAD_FILE_WRITE) {
174 if (uploadFile) uploadFile.write(up.buf, up.currentSize);
175 }
176 else if (up.status == UPLOAD_FILE_END) {
177 if (uploadFile) uploadFile.close();
178 }

202static void apiGetPattern() {
203 String file = D.server->arg("file");
204 if (file.isEmpty()) file = D.cfg->currentPatternFile;
205 file = normalizePatternPath(file);
206
207 bool switchingPattern = (file != D.cfg->currentPatternFile);
208
209 Pattern p;
210 if (!loadPatternFile(file, p)) {
211 // If missing, create from current pattern (or default empty)
212 p = *D.pattern;
213 savePatternFile(file, p);
214 }
215
216 D.cfg->currentPatternFile = file;
217 *D.pattern = p;
218
219 if (switchingPattern) {
220 resetRuntimeState();
221 } else {
222 // keep activeRow valid
223 D.cfg->activeRow = wrapRowIndex(D.cfg->activeRow, D.pattern->h);
224 }
225 saveConfig(*D.cfg);
226
227 String out = "{";
228 out += "\"file\":\"" + htmlEscape(file) + "\",";
229 out += "\"activeRow\":" + String(D.cfg->activeRow) + ",";
230 out += "\"pattern\":" + patternToJson(*D.pattern);
231 out += "}";
232 D.server->send(200, "application/json", out);

235static void apiPostPattern() {
236 String body = D.server->arg("plain");
237
238 // extract "file":"..."
239 int fpos = body.indexOf("\"file\":\"");
240 if (fpos < 0) { D.server->send(400, "text/plain", "Missing file"); return; }
241 fpos += 8;
242 int fend = body.indexOf("\"", fpos);
243 if (fend < 0) { D.server->send(400, "text/plain", "Bad file"); return; }
244 String file = body.substring(fpos, fend);
245 file = normalizePatternPath(file);
246
247 // extract "pattern":{...}
248 int ppos = body.indexOf("\"pattern\":");
249 if (ppos < 0) { D.server->send(400, "text/plain", "Missing pattern"); return; }
250 String pjson = body.substring(ppos + 10);
251 int a = pjson.indexOf("{");
252 int b = pjson.lastIndexOf("}");
253 if (a < 0 || b < 0 || b <= a) { D.server->send(400, "text/plain", "Bad pattern json"); return; }
254 pjson = pjson.substring(a, b + 1);
255
256 Pattern p;
257 if (!jsonToPattern(pjson, p)) { D.server->send(400, "text/plain", "Invalid pattern"); return; }
258 if (!savePatternFile(file, p)) { D.server->send(500, "text/plain", "Write failed"); return; }
259
260 D.cfg->currentPatternFile = file;
261 *D.pattern = p;
262
263 // reset confirmations when pattern changes
264 for (int i = 0; i < MAX_H; i++) D.rowConfirmed[i] = false;
265
266 // keep activeRow valid
267 D.cfg->activeRow = wrapRowIndex(D.cfg->activeRow, D.pattern->h);
268 saveConfig(*D.cfg);
269
270 D.server->send(200, "application/json", "{\"ok\":true}");
271}

360static void apiPostConfig() {
361 String body = D.server->arg("plain");
362
363 auto getNum = [&](const char* key, uint32_t& v) -> bool {
364 String k = String("\"") + key + "\":";
365 int i = body.indexOf(k);
366 if (i < 0) return false;
367 i += k.length();
368 int j = i;
369 while (j < (int)body.length() && (isdigit(body[j]) || body[j] == '-')) j++;
370 v = (uint32_t)body.substring(i, j).toInt();
371 return true;
372 };
373
374 auto getBool = [&](const char* key, bool& v) -> bool {
375 String k = String("\"") + key + "\":";
376 int i = body.indexOf(k);
377 if (i < 0) return false;
378 i += k.length();
379 String t = body.substring(i, min(i + 5, (int)body.length()));
380 if (t.startsWith("true")) v = true;
381 else if (t.startsWith("false")) v = false;
382 else return false;
383 return true;
384 };
385
386 uint32_t ca, cc, ci, br;
387 bool aa, bw, rb;
388
389 if (getNum("colorActive", ca)) D.cfg->colorActive = ca;
390 if (getNum("colorConfirmed", cc)) D.cfg->colorConfirmed = cc;
391 if (getNum("colorInactive", ci)) D.cfg->colorInactive = ci;
392 if (getNum("brightness", br)) D.cfg->brightness = (uint8_t)constrain((int)br, 0, 255);
393
394 if (getBool("autoAdvance", aa)) D.cfg->autoAdvance = aa;
395 if (getBool("blinkWarning", bw)) D.cfg->blinkWarning = bw;
396
397 // New: row counting direction
398 if (getBool("rowFromBottom", rb)) D.cfg->rowFromBottom = rb;
399
400 saveConfig(*D.cfg);
401 D.server->send(200, "application/json", "{\"ok\":true}");
402}
